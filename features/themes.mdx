MiniFy includes 8 built-in themes and supports custom themes via JSON configuration. Themes control colors, shadows, borders, and visual styling across the entire player.

## Built-in Themes

### Dark Theme

The default dark theme with subtle transparency and modern shadows.

```json
{
  "name": "dark",
  "panel": {
    "background": "rgba(18, 18, 18, 0.85)",
    "borderRadius": 18,
    "shadow": "0 14px 36px rgba(0, 0, 0, 0.65)"
  },
  "controls": {
    "iconColor": "#E5E5E5",
    "iconColorActive": "#FFFFFF"
  },
  "playbar": {
    "trackBg": "rgba(255, 255, 255, 0.18)",
    "trackFill": "linear-gradient(90deg, #FFFFFF 0%, #CCCCCC 100%)",
    "thumbColor": "#FFFFFF"
  },
  "typography": {
    "songTitle": { "color": "#FFFFFF", "weight": 600 },
    "songArtist": { "color": "rgba(255, 255, 255, 0.70)", "weight": 400 }
  }
}
```

**Best for:** Night sessions, reduced eye strain, modern aesthetic

## Theme Structure

Every theme JSON file follows this structure:

<CodeGroup>
```json Theme Schema
{
  "name": "theme-name",
  
  "panel": {
    "background": "color | gradient",
    "borderRadius": 18,
    "shadow": "CSS shadow value"
  },
  
  "settings": {
    "panelBg": "rgba(...)",
    "panelBorder": "rgba(...)",
    "text": "#FFFFFF",
    "textMuted": "rgba(...)",
    "itemHover": "rgba(...)",
    "itemActive": "rgba(...)",
    "accent": "#74C7EC"
  },
  
  "controls": {
    "iconColor": "#E5E5E5",
    "iconColorActive": "#FFFFFF",
    "iconBackground": "transparent",
    "iconBackgroundHover": "rgba(...)"
  },
  
  "playbar": {
    "trackBg": "rgba(...)",
    "trackFill": "color | gradient",
    "thumbColor": "#FFFFFF",
    "timeTextColor": "rgba(...)"
  },
  
  "typography": {
    "songTitle": {
      "color": "#FFFFFF",
      "weight": 600
    },
    "songArtist": {
      "color": "rgba(...)",
      "weight": 400
    }
  },
  
  "actions": {
    "iconColor": "#FFFFFF",
    "iconBackground": "rgba(...)",
    "iconBackgroundHover": "rgba(...)"
  },
  
  "cover": {
    "borderColor": "rgba(...)",
    "borderRadius": 12
  }
}
```
</CodeGroup>

## CSS Variables

Themes are applied via CSS custom properties:

```css
/* Panel */
--player-panel-bg
--player-panel-radius
--player-panel-shadow

/* Controls */
--player-controls-color
--player-controls-color-active

/* Playbar */
--player-playbar-track-bg
--player-playbar-track-fill
--player-playbar-thumb-color
--player-playbar-time-color

/* Typography */
--player-song-title-color
--player-song-artist-color

/* Actions */
--player-actions-color
--player-actions-bg
--player-actions-bg-hover

/* Cover */
--player-cover-border-color
--player-cover-radius

/* Settings Panel */
--settings-panel-bg
--settings-panel-border
--settings-text
--settings-text-muted
--settings-header-text
--settings-item-hover
--settings-item-active
--settings-accent
```

## Creating Custom Themes

<Steps>
  <Step title="Create theme JSON">
    Create a new JSON file following the theme structure above
  </Step>
  <Step title="Add to themes directory">
    Place your `custom-theme.json` in the themes directory:
    ```
    ~/MiniFy/themes/custom-theme.json
    ```
  </Step>
  <Step title="Load in Settings">
    Open Settings > Appearance > Custom Themes and import your JSON file
  </Step>
</Steps>

<Tip>
  Start by copying an existing theme and modifying the colors to match your preferences.
</Tip>

## Theme Validation

MiniFy validates custom themes using the `validateThemeJsonFormat` function:

```typescript
export function validateThemeJsonFormat(jsonString: string): ThemeValidationResult {
  try {
    const parsed = JSON.parse(jsonString) as RawTheme;

    // Must have a name
    if (!parsed.name || typeof parsed.name !== "string" || parsed.name.trim() === "") {
      return { valid: false, error: "Theme must have a non-empty 'name' field" };
    }

    // Validate object types
    if (parsed.panel !== undefined && typeof parsed.panel !== "object") {
      return { valid: false, error: "'panel' must be an object" };
    }

    // ... more validation

    return { valid: true };
  } catch (e) {
    return { valid: false, error: "Invalid JSON" };
  }
}
```

## Applying Themes Programmatically

```typescript
import { applyThemeByName, applyCustomThemeFromJson } from "./loader/themeLoader";

// Apply built-in theme
applyThemeByName("dark");

// Apply custom theme
const customThemeJson = JSON.stringify({
  name: "my-theme",
  panel: { background: "#1a1a1a" },
  // ... rest of theme config
});
applyCustomThemeFromJson(customThemeJson);
```

<Note>
  Theme changes apply immediately without requiring a restart.
</Note>
